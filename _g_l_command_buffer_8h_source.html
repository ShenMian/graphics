52template<typename T>
53inline T* GLCommandBuffer::addCommand(GLOpcode opcode)
54{
55 auto offset = buffer.size();
56 buffer.push_back(static_cast<uint8_t>(opcode));
57 buffer.resize(offset + sizeof(GLOpcode) + sizeof(T));
58 return reinterpret_cast<T*>(&buffer[offset + sizeof(GLOpcode)]);
59}
