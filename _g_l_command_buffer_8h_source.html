53template <typename T>
54inline T* GLCommandBuffer::addCommand(GLOpcode opcode)
55{
56 auto offset = buffer.size();
57 buffer.push_back(static_cast<uint8_t>(opcode));
58 buffer.resize(offset + sizeof(GLOpcode) + sizeof(T));
59 return reinterpret_cast<T*>(&buffer[offset + sizeof(GLOpcode)]);
60}
